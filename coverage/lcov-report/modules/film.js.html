    const search = {
        method: 'GET'
        , url: 'http://www.omdbapi.com/'
        , qs: {
            s: movieSearch
            , r: 'json'
        }
    , }
    request(search, function (error, response, body) {
        if (error) throw new Error(error)
        return callback(null, JSON.parse(body))
    })
}

}
